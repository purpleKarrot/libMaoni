include(Extract)
include(PurpleLibrary)

set(QT_USE_QTXML ON)
set(QT_USE_QTOPENGL ON)
find_package(Qt4 REQUIRED)
include(${QT_USE_FILE})

if(WIN32)
  set(Boost_USE_STATIC_LIBS ON)
endif(WIN32)

find_package(Boost  REQUIRED COMPONENTS serialization)
find_package(OpenGL REQUIRED)

extract(boost-la.tar.gz ${CMAKE_BINARY_DIR}/include)

include_directories(
  ${Boost_INCLUDE_DIRS}
  ${OPENGL_INCLUDE_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR}
  ${CMAKE_BINARY_DIR}/include
  ${CMAKE_CURRENT_BINARY_DIR}/QGLViewer
  ${CMAKE_CURRENT_BINARY_DIR}/QPropertyBrowser
  )

add_definitions(
  -DBOOST_ALL_NO_LIB
  -DQGLVIEWER_STATIC
  )


add_subdirectory(Common)
add_subdirectory(QGLViewer)
add_subdirectory(QPropertyBrowser)

set(MAONI_LIBRARIES
  Common
  QGLViewer
  QtPropertyBrowser
  ${QT_LIBRARIES}
  ${Boost_LIBRARIES}
  ${OPENGL_LIBRARIES}
  )

if(MSVC)
  set(MAONI_VERSION_RC ${CMAKE_CURRENT_SOURCE_DIR}/version.rc)
endif(MSVC)


#add_subdirectory(Equalizer)
add_subdirectory(IceT)


purple_library(Maoni SHARED
  SOURCES Maoni.cpp ${MAONI_VERSION_RC}
  DEPENDS ${MAONI_LIBRARIES}
  )

purple_library(MaoniMain STATIC
  SOURCES main.cpp
  )
